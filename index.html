// --- Yeh hissa rates.json ko app se joddta hai --- async function loadRates() { try { // new Date().getTime() ki wajah se browser hamesha fresh file mangta hai const response = await fetch('rates.json?v=' + new Date().getTime()); if (!response.ok) throw new Error('Rates file nahi mili'); AAJ_KA_RATE = await response.json(); // File se data uthakar app mein set karta hai updateUI(); // Screen par rate update karta hai } catch (error) { console.error("Rates error:", error); // Agar file na mile toh yeh purane rates dikha dega taaki app khali na rahe AAJ_KA_RATE = { whole: 180, curry: 220, bone: 350 }; updateUI(); } }
